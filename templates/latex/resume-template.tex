\newcommand{\resumeItem}[1]{
  \item\small{
    {#1 \vspace{-1.5pt}}
  }
}

% Generic 4-field heading: {left top}{right top}{left bottom}{right bottom}
% Kept as a fallback; Education now uses \resumeExpHeading so the degree sits
% bold alongside the institution.
\newcommand{\resumeSubheading}[4]{
  \vspace{-1pt}\item
    \begin{tabular*}{0.97\textwidth}[t]{l@{\extracolsep{\fill}}r}
      \textbf{#1} & #2 \\
      \textit{\small#3} & \textit{\small #4} \\
    \end{tabular*}\vspace{-7pt}
}

% Experience/Awards heading, one line:
% {Company}{Role}{Dates}{Location}
\newcommand{\resumeExpHeading}[4]{
  \vspace{1pt}\item
    \begin{tabular*}{0.97\textwidth}[t]{l@{\extracolsep{\fill}}r}
      \textbf{#1} {\normalfont\small$\vert$} \textbf{\textit{#2}} & {\small \textit{#3} $\vert$ #4} \\
    \end{tabular*}\vspace{-7pt}
}

% Second role at the same employer: {Role}{Dates}
\newcommand{\resumeSubSubheading}[2]{
    \item
    \begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
      \textit{\small#1} & \textit{\small #2} \\
    \end{tabular*}\vspace{-7pt}
}

% Project heading: {Name (+ context)}{Stack or dates}
\newcommand{\resumeProjectHeading}[2]{
    \item
    \begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
      \small#1 & #2 \\
    \end{tabular*}\vspace{-7pt}
}

%-----------SUMMARY-----------
% 3-5 lines. A portrait of the candidate, never a paraphrase of the JD:
% profile (the roles actually operated in), academic history, then intent
% capped with an extremely concise why-fit clause. Add any hard constraint
% the recruiter must know (location, visa, notice). Bold sparingly.
\section{Summary}
\vspace{2pt}
\small{<PROFILE: the role(s) the candidate has operated in — research,
engineering, a mix — and at what level>. <ACADEMIC HISTORY: degrees and, when
they carry weight, institutions>. <INTENT: what they want next and where,
e.g. thrive in <CITY>'s rich AI ecosystem>, <WHY-FIT: extremely concise,
finishing the intent sentence>. <HARD CONSTRAINT, if any: location, work
authorisation, availability>.}
\vspace{-2pt}

%-----------SKILLS-----------
% One block: 4-6 main abilities in bold, each followed by its keywords and
% tools. The ATS harvest lives inside these groups — verbatim JD spelling,
% every term evidenced in knowledge/. Order the groups by what the JD leads
% with; within a group, most-relevant tool first; years only where depth sells.
\section{Skills}
 \begin{itemize}[leftmargin=0.15in, label={}]
    \footnotesize{\item{
     \textbf{<ABILITY 1 — e.g. Infrastructure \& Languages>}{: <TOOL> (<N>+ yrs), <TOOL>, <TOOL>; <TOOL>, <TOOL>, <TOOL>} \\
     \textbf{<ABILITY 2 — e.g. ML \& Generative AI>}{: <TOOL>, <TOOL>, <TOOL>, <TOOL>} \\
     \textbf{<ABILITY 3 — e.g. Data \& Pipelines>}{: <TOOL>, <TOOL>, <TOOL>, <TOOL>} \\
     \textbf{<ABILITY 4 — e.g. Evaluation \& Observability>}{: <TOOL>, <TOOL>, <TOOL>, <TOOL>}
    }}
 \end{itemize}
\vspace{-4pt}

%-----------PROJECTS-----------
% Thesis work, research, and side projects that show depth the jobs don't.
% Best-matching projects for the JD first. Where space allows, be
% comprehensive: what was done and achieved, the engineering techniques

% relevant project may take more than one bullet; balance that depth against
